// Um Camarão Orgulhoso - 2025 Edition 🦐 const canvas = document.createElement('canvas'); canvas.width = 600; canvas.height = 400; canvas.style.background = 'linear-gradient(to bottom, #0f3460, #1a5aa6)'; document.body.appendChild(canvas); const ctx = canvas.getContext('2d'); // Variáveis do nosso camarão famoso let camarao = { x: 300, y: 200, tamanho: 50, velocidadeX: 0, velocidadeY: 0, antenasBalançando: 0, feliz: true }; function desenharCamarao() { ctx.save(); ctx.translate(camarao.x, camarao.y); // Corpo principal (rosado/laranja) ctx.fillStyle = '#ff6b6b'; ctx.beginPath(); ctx.ellipse(0, 0, camarao.tamanho, camarao.tamanho * 0.6, 0, 0, Math.PI * 2); ctx.fill(); // Cabeça um pouco mais clara ctx.fillStyle = '#ff8787'; ctx.beginPath(); ctx.ellipse(-camarao.tamanho * 0.35, -5, camarao.tamanho * 0.55, camarao.tamanho * 0.5, 0, 0, Math.PI * 2); ctx.fill(); // Olhos ctx.fillStyle = 'white'; ctx.beginPath(); ctx.arc(-camarao.tamanho * 0.45, -camarao.tamanho * 0.25, 12, 0, Math.PI * 2); ctx.arc(-camarao.tamanho * 0.45, camarao.tamanho * 0.05, 12, 0, Math.PI * 2); ctx.fill(); ctx.fillStyle = 'black'; ctx.beginPath(); ctx.arc(-camarao.tamanho * 0.45, -camarao.tamanho * 0.25, 5, 0, Math.PI * 2); ctx.arc(-camarao.tamanho * 0.45, camarao.tamanho * 0.05, 5, 0, Math.PI * 2); ctx.fill(); // Antenas balançando const anguloAntena = Math.sin(camarao.antenasBalançando) * 0.4; ctx.strokeStyle = '#e65c5c'; ctx.lineWidth = 4; // Antena esquerda ctx.beginPath(); ctx.moveTo(-camarao.tamanho * 0.6, -camarao.tamanho * 0.4); ctx.quadraticCurveTo( -camarao.tamanho * 0.9 + Math.cos(anguloAntena) * 20, -camarao.tamanho * 0.9, -camarao.tamanho * 1.1, -camarao.tamanho * 0.7 + Math.sin(anguloAntena) * 30 ); ctx.stroke(); // Antena direita ctx.beginPath(); ctx.moveTo(-camarao.tamanho * 0.6, camarao.tamanho * 0.2); ctx.quadraticCurveTo( -camarao.tamanho * 0.9 - Math.cos(anguloAntena) * 20, camarao.tamanho * 0.7, -camarao.tamanho * 1.1, camarao.tamanho * 0.4 + Math.sin(anguloAntena) * 30 ); ctx.stroke(); // Rabinho balançando (o famoso rabinho do camarão) const rabinhoAng = Math.sin(camarao.antenasBalançando * 1.8) * 0.5; ctx.fillStyle = '#ff5252'; ctx.beginPath(); ctx.moveTo(camarao.tamanho * 0.8, 0); ctx.quadraticCurveTo( camarao.tamanho * 1.3, -camarao.tamanho * 0.4 * rabinhoAng, camarao.tamanho * 1.1, -camarao.tamanho * 0.7 * rabinhoAng ); ctx.quadraticCurveTo( camarao.tamanho * 0.9, 0, camarao.tamanho * 1.1, camarao.tamanho * 0.7 * rabinhoAng ); ctx.quadraticCurveTo( camarao.tamanho * 1.3, camarao.tamanho * 0.4 * rabinhoAng, camarao.tamanho * 0.8, 0 ); ctx.closePath(); ctx.fill(); ctx.restore(); } function animar() { ctx.clearRect(0, 0, canvas.width, canvas.height); // Movimento suave aleatório camarao.velocidadeX += (Math.random() - 0.5) * 0.4; camarao.velocidadeY += (Math.random() - 0.5) * 0.4; // Limitar velocidade camarao.velocidadeX = Math.max(-3, Math.min(3, camarao.velocidadeX)); camarao.velocidadeY = Math.max(-3, Math.min(3, camarao.velocidadeY)); camarao.x += camarao.velocidadeX; camarao.y += camarao.velocidadeY; // Não sair da tela camarao.x = Math.max(camarao.tamanho * 1.5, Math.min(canvas.width - camarao.tamanho * 1.5, camarao.x)); camarao.y = Math.max(camarao.tamanho, Math.min(canvas.height - camarao.tamanho, camarao.y)); // Balançar antenas e rabinho camarao.antenasBalançando += 0.08; desenharCamarao(); // Textinho fofo ctx.fillStyle = 'white'; ctx.font = '22px Arial'; ctx.textAlign = 'center'; ctx.fillText("🦐 O Camarão Mais Estiloso do Oceano 🦐", canvas.width/2, 40); requestAnimationFrame(animar); } // Começa a zoeira animar(); // Clica pra dar um susto no camarão (ele foge rapidão) canvas.addEventListener('click', (e) => { const rect = canvas.getBoundingClientRect(); const mouseX = e.clientX - rect.left; const mouseY = e.clientY - rect.top; const dx = camarao.x - mouseX; const dy = camarao.y - mouseY; const distancia = Math.sqrt(dx*dx + dy*dy); if (distancia < camarao.tamanho * 2) { camarao.velocidadeX = dx * 0.3; camarao.velocidadeY = dy * 0.3; // Piscadinha de medo setTimeout(() => camarao.feliz = true, 400); camarao.feliz = false; } });
